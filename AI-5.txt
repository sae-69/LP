def chatbot():
    print("Hi, I am a chatbot. How can I help you today?")
    
    while True:
        user_input = input("You: ")
        
        if user_input.lower() == "bye":
            print("Chatbot: Bye, have a great day!")
            break
        
        elif user_input.lower() == "hello":
            print("Chatbot: Hello! How are you today?")
        
        elif user_input.lower() == "how are you":
            print("Chatbot: I am doing great, thanks for asking. What would you like to see?")
        
        elif user_input.lower() == "phone":
            print("Chatbot: Which brand?")
        
        elif user_input.lower() == "apple":
            print("Chatbot: Ok. What is your budget?")
        
        elif user_input.lower() == "100000":
            print("Chatbot: Ok, here are some top models.")
        
        elif user_input.lower() == "thank you":
            print("Chatbot: You're welcome!\nSee you again.")
        
        else:
            print("Chatbot: I did not understand what you said. Sorry.")

chatbot()
